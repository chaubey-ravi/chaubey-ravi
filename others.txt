Sub CopyDataToDestinationWorkbook()
    Dim destWorksheet As Worksheet
    Dim sourceFile As Variant
    Dim sourceWorkbook As Workbook
    Dim destLastRow As Long
    
    ' Step 2: Activate "OD by Site" tab, remove filter, unhide columns
    Set destWorksheet = destWorkbook.Sheets("OD by Site")
    With destWorksheet
        .Activate
        .AutoFilterMode = False
        .Columns.Hidden = False
    End With
    
    ' Step 3: Clear data in columns A, B to L, M to N
    With destWorksheet
        ' Clear data in column A from row 3 to the last row
        .Range("A3:A" & .Cells(.Rows.Count, "A").End(xlUp).Row).ClearContents
        
        ' Clear data from column B to L starting from row 2 to the last row
        .Range("B2:L" & .Cells(.Rows.Count, "B").End(xlUp).Row).ClearContents
        
        ' Clear data from column M to N from row 3 to the last row
        .Range("M3:N" & .Cells(.Rows.Count, "M").End(xlUp).Row).ClearContents
    End With
    
    ' Step 4: Ask user to select source file
    sourceFile = Application.GetOpenFilename("Excel Files (*.xls; *.xlsx), *.xls; *.xlsx")
    
    ' Check if the user canceled the file selection
    If sourceFile = "False" Then Exit Sub
    
    ' Open source workbook
    Set sourceWorkbook = Workbooks.Open(sourceFile)
    
    ' Step 8: Activate first worksheet, remove filter if there is any
    With sourceWorkbook.Sheets(1)
        .Activate
        .AutoFilterMode = False
    End With
    
    ' Step 9: Copy data from source file and paste only values to destination file
    destLastRow = destWorksheet.Cells(destWorksheet.Rows.Count, "A").End(xlUp).Row
    
    ' Copy and paste data one by one
    destWorksheet.Range("B2:B" & destLastRow).Value = sourceWorkbook.Columns("K").Value
    destWorksheet.Range("C2:C" & destLastRow).Value = sourceWorkbook.Columns("R").Value
    destWorksheet.Range("D2:D" & destLastRow).Value = sourceWorkbook.Columns("AD").Value
    destWorksheet.Range("E2:E" & destLastRow).Value = sourceWorkbook.Columns("GL").Value
    destWorksheet.Range("F2:F" & destLastRow).Value = sourceWorkbook.Columns("AF").Value
    destWorksheet.Range("G2:G" & destLastRow).Value = sourceWorkbook.Columns("AG").Value
    destWorksheet.Range("H2:H" & destLastRow).Value = sourceWorkbook.Columns("FH").Value
    destWorksheet.Range("I2:I" & destLastRow).Value = sourceWorkbook.Columns("CS").Value
    destWorksheet.Range("J2:J" & destLastRow).Value = sourceWorkbook.Columns("EP").Value
    destWorksheet.Range("K2:K" & destLastRow).Value = sourceWorkbook.Columns("EQ").Value
    destWorksheet.Range("L2:L" & destLastRow).Value = sourceWorkbook.Columns("FR").Value
    
    ' Step 10: Fill down formula till the last row for column A, M, and N
    With destWorksheet
        .Range("A3:A" & destLastRow).Formula = .Range("A2").Formula
        .Range("M3:M" & destLastRow).Formula = .Range("M2").Formula
        .Range("N3:N" & destLastRow).Formula = .Range("N2").Formula
    End With
    
    ' Close source workbook without saving changes
    sourceWorkbook.Close SaveChanges:=False
End Sub
