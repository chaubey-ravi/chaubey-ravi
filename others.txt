Sub CopyDataToDestinationWorkbook()
    Dim destFile As Variant
    Dim sourceFile As Variant
    Dim destWorkbook As Workbook
    Dim sourceWorkbook As Workbook
    
    ' Step 1: Ask user to select destination file
    destFile = Application.GetOpenFilename("Excel Files (*.xls; *.xlsx), *.xls; *.xlsx")
    
    ' Check if the user canceled the file selection
    If destFile = "False" Then Exit Sub
    
    ' Open destination workbook
    Set destWorkbook = Workbooks.Open(destFile)
    
    ' Step 2: Activate "QMI (sere)" tab, remove filters, and unhide columns
    With destWorkbook.Sheets("QMI (sere)")
        .Activate
        .AutoFilterMode = False
        .Columns.Hidden = False
    End With
    
    ' Step 3: Clear data from column A to DF till the last row
    With destWorkbook.Sheets("QMI (sere)").Columns("A:DF")
        .ClearContents
    End With
    
    ' Step 4: Ask user to select QMI workbook
    sourceFile = Application.GetOpenFilename("Excel Files (*.xls; *.xlsx), *.xls; *.xlsx")
    
    ' Check if the user canceled the file selection
    If sourceFile = "False" Then
        destWorkbook.Close SaveChanges:=False
        Exit Sub
    End If
    
    ' Open QMI workbook
    Set sourceWorkbook = Workbooks.Open(sourceFile)
    
    ' Step 5: Activate first sheet, remove filter if there is any
    With sourceWorkbook.Sheets(1)
        .Activate
        .AutoFilterMode = False
    End With
    
    ' Step 6: Select data from column A to DF and copy it
    With sourceWorkbook.Sheets(1).Columns("A:DF")
        .Copy
    End With
    
    ' Step 7: Paste data in destination file "QMI (sere)" tab at cell A1, paste values only
    destWorkbook.Sheets("QMI (sere)").Range("A1").PasteSpecial Paste:=xlPasteValues
    
    ' Close source workbook without saving changes
    sourceWorkbook.Close SaveChanges:=False
    
    ' Close destination workbook and save changes
    destWorkbook.Close SaveChanges:=True
End Sub
