Sub FillDownFormula()
    Dim lastRow As Long
    
    ' Find the last row with data in column A
    lastRow = Cells(Rows.Count, 1).End(xlUp).Row
    
    ' Check if there is data in column A
    If lastRow > 1 Then
        ' Replace "A2" with the cell reference where your formula starts
        Range("A2").AutoFill Destination:=Range("A2:A" & lastRow), Type:=xlFillDefault
    Else
        MsgBox "No data found in column A", vbExclamation
    End If
End Sub
